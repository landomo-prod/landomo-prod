🔧 EXAMPLE USAGE
================================================================================

const TopRealityScraper = require('./topreality_working_scraper');

(async () => {
    const scraper = new TopRealityScraper();
    await scraper.init();

    // Get all regions
    const regions = await scraper.getAllRegions();

    // Get property count for Bratislava apartments for sale
    const count = await scraper.getPropertyCount({
        location: 'c100-Bratislavský+kraj',
        offerType: '1',  // Sale
        propertyType: '1' // Apartments
    });

    console.log(`Found ${count} apartments for sale in Bratislava`);
    // Output: Found 9,200 apartments for sale in Bratislava

    await scraper.close();
})();
